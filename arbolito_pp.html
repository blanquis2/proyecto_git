arbolito<-rtree(20)

ggtree(arbolito)+
  geom_tiplab(colour="bisque3")+
  geom_nodepoint(color="#b5e521", alpha=1/4, size=10)+ 
  geom_tippoint(color="#FDAC4F", shape=8, size=3)
